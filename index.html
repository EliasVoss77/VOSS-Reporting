export default { async fetch(request, env) { const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type, x-api-key" }; if (request.method === "OPTIONS") { return new Response(null, { headers: corsHeaders }); } const url = new URL(request.url); // Share link fetcher endpoint if (request.method === 'POST' && url.pathname === '/fetch-share') { const apiKey = request.headers.get("x-api-key"); if (apiKey !== "voss-dk7x-2f9m-q372") { return new Response("Unauthorized", { status: 401, headers: corsHeaders }); } try { const body = await request.json(); const shareUrl = body.url || ''; // Try multiple fetch strategies let pageText = ''; let fetchError = ''; // Strategy 1: Direct fetch with browser-like headers try { const direct = await fetch(shareUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36', 'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8', 'Accept-Language': 'en-US,en;q=0.5', } }); if (direct.ok) { pageText = await direct.text(); } else { fetchError = 'Direct fetch failed: ' + direct.status; } } catch(e) { fetchError = 'Direct fetch error: ' + e.message; } // Strategy 2: Try Jina if direct failed if (!pageText || pageText.length < 100) { try { const jina = await fetch('https://r.jina.ai/' + shareUrl, { headers: { 'Accept': 'text/plain', 'X-Return-Format': 'text' } }); if (jina.ok) { pageText = await jina.text(); } else { fetchError += ' | Jina failed: ' + jina.status; } } catch(e) { fetchError += ' | Jina error: ' + e.message; } } if (!pageText || pageText.length < 100) { return new Response(JSON.stringify({ error: 'Could not fetch page. ChatGPT share links require JavaScript rendering. Try: copy the conversation text and paste it directly, or export and upload the JSON file. Debug: ' + fetchError }), { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // Use Claude to extract conversation const extractResponse = await fetch("https://api.anthropic.com/v1/messages", { method: "POST", headers: { "Content-Type": "application/json", "x-api-key": env.ANTHROPIC_API_KEY, "anthropic-version": "2023-06-01" }, body: JSON.stringify({ model: "claude-haiku-4-5-20251001", max_tokens: 8000, messages: [{ role: "user", content: `Extract the conversation from this page content. Format as: User: [message] Assistant: [message] Return ONLY the conversation, no commentary. Content: ${pageText.substring(0, 50000)}` }] }) }); const extractData = await extractResponse.json(); const text = extractData.content?.[0]?.text || ''; if (!text || text.length < 20) { return new Response(JSON.stringify({ error: 'Could not parse conversation from this page' }), { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } return new Response(JSON.stringify({ text }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } catch (err) { return new Response(JSON.stringify({ error: err.message }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } } // PDF text extraction endpoint if (request.method === 'POST' && url.pathname === '/extract-pdf') { const apiKey = request.headers.get("x-api-key"); if (apiKey !== "voss-dk7x-2f9m-q372") { return new Response("Unauthorized", { status: 401, headers: corsHeaders }); } try { const body = await request.json(); const pdfBase64 = body.pdf || ''; const extractResponse = await fetch("https://api.anthropic.com/v1/messages", { method: "POST", headers: { "Content-Type": "application/json", "x-api-key": env.ANTHROPIC_API_KEY, "anthropic-version": "2023-06-01" }, body: JSON.stringify({ model: "claude-haiku-4-5-20251001", max_tokens: 8000, messages: [{ role: "user", content: [ { type: "document", source: { type: "base64", media_type: "application/pdf", data: pdfBase64 } }, { type: "text", text: "Extract all text content from this PDF document. Do not add any commentary or formatting." } ] }] }) }); const extractData = await extractResponse.json(); const text = extractData.content?.[0]?.text || ''; return new Response(JSON.stringify({ text }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } catch (err) { return new Response(JSON.stringify({ error: err.message }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } } if (request.method !== "POST") { return new Response("Method not allowed", { status: 405, headers: corsHeaders }); } const apiKey = request.headers.get("x-api-key"); if (apiKey !== "voss-dk7x-2f9m-q372") { return new Response("Unauthorized", { status: 401, headers: corsHeaders }); } try { const body = await request.json(); const chatText = body.chat || ""; const name = body.name || "Unknown"; const department = body.department || "Unknown"; if (chatText.trim().length === 0) { return new Response("No chat text provided", { status: 400, headers: corsHeaders }); } const prompt = ` You are an AI productivity analyst for VOSS Reporting.
